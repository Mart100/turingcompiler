5 2 0 0 1 0 0 1 1 0 3 0 0 0 0 1 1 1 0 4

# Start of the addition program
START 5 5 R FIND_SECOND_END

# Find the end of the second number
FIND_SECOND_END 0 0 R FIND_SECOND_END
FIND_SECOND_END 1 1 R FIND_SECOND_END
FIND_SECOND_END 2 2 R FIND_SECOND_END
FIND_SECOND_END 3 3 R FIND_SECOND_END
FIND_SECOND_END 4 4 L SUBTRACT_ONE

# Subtract one from the second number
SUBTRACT_ONE 0 1 L SUBTRACT_ONE
SUBTRACT_ONE 1 0 L FIND_FIRST_END
SUBTRACT_ONE 3 3 L TO_MIDDLE

# Find the end of the first number. Move left
FIND_FIRST_END 0 0 L FIND_FIRST_END
FIND_FIRST_END 1 1 L FIND_FIRST_END
FIND_FIRST_END 3 3 L ADD_ONE

# Add one to the first number
ADD_ONE 0 1 L FIND_SECOND_END
ADD_ONE 1 0 L ADD_ONE
ADD_ONE 3 3 L TO_MIDDLE

# Return to the middle of the tape
TO_MIDDLE 0 0 L TO_MIDDLE
TO_MIDDLE 1 1 L TO_MIDDLE
TO_MIDDLE 2 2 L TO_MIDDLE
TO_MIDDLE 5 5 S END
